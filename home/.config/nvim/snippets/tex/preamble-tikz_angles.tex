\newcommand{\MarkRightAngle}[5][]{
\draw[#1] let \p1=($(#4)-(#3)$), \p2=($(#5)-(#4)$), \p0=($0*(\p1)$) in
(#4)--($(#4)!#2!(#3)$)--++($(\p0)!#2!(\p2)$)--++($(\p0)!#2!(\p1)$)--cycle;}

\newcommand{\MarkAngle}[5]["$\theta$",->,draw=blue!80,fill=blue!20]{
\pic[#1,angle radius=#2] {angle = #3--#4--#5};}
