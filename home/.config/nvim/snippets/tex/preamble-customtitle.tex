%%% Default Title %%%
\makeatletter\def\@maketitle{%
\newpage\null\vskip2em%
\begin{center}%
	\let\footnote\thanks%
	{\LARGE\@title\par}%
	\vskip1.5em%
	{\large\lineskip.5em%
	\begin{tabular}[t]{c}%
		\@author%
	\end{tabular}%
	\par}%
	\vskip1em%
	{\large\@date}%
\end{center}%
\par\vskip1.5em}
\makeatother
%-----------------------------------------------------------------------------%
%%% Minimal Title %%%
\makeatletter
\newsavebox{\titledatebox}
\def\@maketitle{%
\begingroup
\sbox{\titledatebox}{{\bfseries\@title}\quad\texttt{\@date}}
\noindent
\ifdim\wd\titledatebox>\textwidth
{\raggedright\bfseries\@title\par}{\raggedleft\texttt{\@date}\par}
\else
{\raggedright\noindent\bfseries\@title}\hfill\texttt{\@date}\par
\fi
{\raggedleft\ttfamily\@author\par}\vskip1em\endgroup
}\makeatother
