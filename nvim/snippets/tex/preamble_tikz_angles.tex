%%% TikZ: Mark Angles %%%
\newcommand{\MarkRightAngle}[5][]{
\draw[#1] let
\p1=($(#3)-(#4)$),\n1={atan2(\y1,\x1)},
\p2=($(#5)-(#4)$),\n2={atan2(\y2,\x2)} in
(#4) -- ++(\n1:#2) -- ++(\n2:#2) -- ++({\n1+180}:#2) --cycle;}
%-----------------------------------------------------------------------------%
\newcommand{\MarkAngle}[6][]{
\path[#1] let
\p1=($(#3)-(#4)$),\n1={atan2(\y1,\x1)},
\p2=($(#5)-(#4)$),\n2={atan2(\y2,\x2)} in
(#4) -- ++($#2*($(#4)!1cm!(#3)$)$) arc (\n1:\n2:#2) #6;}
