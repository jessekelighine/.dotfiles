%%% Default Title %%%
\makeatletter\def\@maketitle{%
\newpage\null\vskip2em%
\begin{center}%
	\let\footnote\thanks%
	{\LARGE\@title\par}%
	\vskip1.5em%
	{\large\lineskip.5em%
	\begin{tabular}[t]{c}%
		\@author%
	\end{tabular}%
	\par}%
	\vskip1em%
	{\large\@date}%
\end{center}%
\par\vskip1.5em}
\makeatother

\noindent{\sffamily\bfseries\@title}%
\hfill\texttt{\@date}\\{\ }\null\hfill\texttt{\@author}
\par\vskip1.5em}\makeatother
